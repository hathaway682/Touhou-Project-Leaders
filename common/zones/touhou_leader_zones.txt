zone_garden = {
	icon = GFX_zone_garden
	base_buildtime = @zone_buildtime

	potential = { # planet scope
		exists = owner
		owner = {
			any_owned_leader = {
				OR = {
					has_trait = thl_leader_trait_flower
					has_trait = thl_leader_trait_flower_commander
				}
			}
		}
	}

	unlock = { # planet scope
		exists = owner
		owner = {
			any_owned_leader = {
				OR = {
					has_trait = thl_leader_trait_flower
					has_trait = thl_leader_trait_flower_commander
				}
			}
		}
	}
	resources = {
		category = planet_zones
		cost = {
			minerals = @zone_cost
		}
	}

	inline_script = {
		script = jobs/zone_growers_add
		AMOUNT = @scaling_district_1_job
	}

	zone_sets = {
		food
	}

	included_building_sets = {
		farming
		hydroponics
		automation
		fallen_empire_farming
		flower
	}

	planet_modifier = {
		zone_building_slots_add = 3
	}

	triggered_desc = {
		trigger = {
			always = yes
		}
		text = zone_food_triggered_desc
	}

	convert_to = {
		zone_rw_garden
	}
}

		OR = {
